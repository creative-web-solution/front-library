Example
let validator = new Validator( $form, options )

on( $form, {
     "eventsName": "submit",
     "callback": e => {
        e.preventDefault();
        validator
                .validate()
                .then( data => {
                        console.log( 'Valid', data );
                        $form.submit();
                    }
                )
                .catch( data => {
                        console.log( 'Invalid', data );
                        data.errors.forEach( input => {
                                console.log( input.getErrorMessages() );
                            }
                        )
                    }
                );
    }
});
