Example
let sn = new ScrollSnap($scroller, options);

// To refresh
sn.refresh( [options] )

// To navigate
sn.previous( [duration] )
sn.next( [duration] )
sn.scrollToItem( $item [, duration] )
sn.scrollToIndex( index [, duration] )

// To remove
sn.clean()
